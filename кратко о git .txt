git status - статус изменений в репазитории
git add . / git add "name fael" - давовление к изменению 
git commit -m "название изменений" - закрипление изменений в git

git branch - просмотр ветак репазитория
git branch названиеВетки - создание новой ветки
git branch -D названиеВетки - удоление ветки
git checkout названиеВетки - переход на ветку
git checkout -b названиеВетки - создание новой ветки и переход на неё
git marge названиеВетки - добавления изменений одной ветки в текущую ветку (где находимся сечас)
git push - заливаем на github
git pull - загрузка обнавлений из github на локальный репазиторий

mkdir название папки - зоздание новой папки
cd путь/путь - переход в нужную папку
git clone ссылка с github на репазиторий - сохранение репазитория на комп 

.gitignore - фаел в котором пишется то что должно игнарироваться g
